Definieren einer Tabelle Mysql
++

CREATE  TABLE `Bewerbungen`.`Arbeitgeber` (
  `ID` INT NOT NULL ,
  `Name` VARCHAR(60) NULL ,
  `Strasse` VARCHAR(120) NULL ,
  `Ort` VARCHAR(45) NULL ,
  `PLZ` DECIMAL(5) NULL ,
  PRIMARY KEY (`ID`) ,
  INDEX `fk_Arbeitgeber_1` () );

++
